## üß© How to Build Yourself (Guideline / Code Outline)

Below is a **simple node.js example** illustrating pack generation + inventory tracking based on rarities.

### Example: Card Pool & Pack Generation (Node.js)

```js
// Example deck data
const cards = [
  { id: 'C001', name: 'Common A', rarity: 'common' },
  { id: 'C002', name: 'Common B', rarity: 'common' },
  { id: 'U001', name: 'Uncommon A', rarity: 'uncommon' },
  { id: 'R001', name: 'Rare A', rarity: 'rare' },
  // ‚Ä¶ add more
];

// Rarity weights for pack logic
const rarityWeights = {
  common: 70,
  uncommon: 20,
  rare: 9,
  ultra: 1
};

function getRandomCardByRarity() {
  const total = Object.values(rarityWeights).reduce((a, b) => a + b, 0);
  let rand = Math.random() * total;
  for (let rarity in rarityWeights) {
    rand -= rarityWeights[rarity];
    if (rand <= 0) {
      const pool = cards.filter(c => c.rarity === rarity);
      return pool[Math.floor(Math.random() * pool.length)];
    }
  }
}

// Generate pack with e.g., 10 cards
function generatePack(size = 10) {
  const pack = [];
  while (pack.length < size) {
    const card = getRandomCardByRarity();
    if (!pack.find(p => p.id === card.id)) { // no duplicates
      pack.push(card);
    }
  }
  return pack;
}

// Track inventory
let inventory = {};
function addToInventory(card) {
  inventory[card.id] = (inventory[card.id] || 0) + 1;
}

// Example usage:
const pack = generatePack(10);
pack.forEach(c => addToInventory(c));
console.log('Pack:', pack);
console.log('Inventory counts:', inventory);
```
